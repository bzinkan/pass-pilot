1) Database: make “null” impossible
Give every PK a default and mark required cols NOT NULL.

Migration SQL (run once):

sql
Copy
Edit
-- UUIDs for all tables
CREATE EXTENSION IF NOT EXISTS pgcrypto;

-- Example: passes
ALTER TABLE passes
  ALTER COLUMN id SET DATA TYPE uuid USING gen_random_uuid(),
  ALTER COLUMN id SET DEFAULT gen_random_uuid(),
  ALTER COLUMN id SET NOT NULL;

-- Required FKs
ALTER TABLE passes ALTER COLUMN school_id SET NOT NULL;
ALTER TABLE passes ALTER COLUMN student_id SET NOT NULL;

-- Timestamps that auto-fill
ALTER TABLE passes
  ALTER COLUMN created_at SET DEFAULT now(),
  ALTER COLUMN created_at SET NOT NULL;
Optional “empty string is invalid” checks (prevents "", which isn’t NULL):

2) Drizzle schema: defaults + .notNull()
ts
Copy
Edit
import { pgTable, uuid, text, timestamp } from "drizzle-orm/pg-core";
import { schools } from "./schools";
import { students } from "./students";

export const passes = pgTable("passes", {
  id: uuid("id").defaultRandom().primaryKey(),              // DB default too
  schoolId: uuid("school_id").notNull()
    .references(() => schools.id, { onDelete: "cascade" }),
  studentId: uuid("student_id").notNull()
    .references(() => students.id, { onDelete: "cascade" }),
  reason: text("reason"),                                   // optional; add .notNull() if required
  createdAt: timestamp("created_at", { withTimezone: true }).defaultNow().notNull(),
});

Null/unknown key stripper middleware (drop-in):

ts
Copy
Edit
// middleware/sanitize.ts
export function sanitizeBody(req, _res, next) {
  const clean = (obj: any) =>
    Object.fromEntries(
      Object.entries(obj ?? {})
        .filter(([k, v]) => v !== null && v !== undefined && k !== "id") // drop id + nulls
        .map(([k, v]) => [k, typeof v === "string" ? v : v])
    );
  if (req.body && typeof req.body === "object") req.body = clean(req.body);
  next();
}

Copy
Edit
import { db } from "../db";
import { passes } from "../db/schema";
import { CreatePassSchema } from "../validation/passSchemas";
import crypto from "node:crypto";

app.post("/api/passes", async (req, res, next) => {
  try {
    const input = CreatePassSchema.parse(req.body); // 422 if bad
    const inserted = await db.insert(passes).values({
      id: crypto.randomUUID(),         // or omit and let DB default fill it
      ...input,
    }).returning();
    res.status(201).json(inserted[0]);
  } catch (err) { next(err); }
});
4) Friendly errors instead of raw 400s
Map Postgres constraint codes → human messages.

ts
Copy
Edit
// middleware/errorHandler.ts
export function errorHandler(err, _req, res, _next) {
  // Zod
  if (err?.issues) return res.status(422).json({ message: "Invalid input", issues: err.issues });

  // Postgres not-null violation
  if (err?.code === "23502") {
    return res.status(400).json({
      message: `Missing required field: ${err.column || "unknown"}`,
      hint: "Check request body and DB defaults.",
    });
  }
  // FK violation (bad schoolId/studentId)
  if (err?.code === "23503") {
    return res.status(400).json({ message: "Related record not found (foreign key)." });
  }
  console.error(err);
  res.status(500).json({ message: "Unexpected server error" });
}
