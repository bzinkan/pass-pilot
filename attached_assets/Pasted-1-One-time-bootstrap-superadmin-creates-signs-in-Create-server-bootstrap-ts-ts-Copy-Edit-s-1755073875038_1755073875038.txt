// server/bootstrap.ts
import type { Express, Request, Response } from "express";
import bcrypt from "bcryptjs";
import pkg from "pg";
const { Pool } = pkg;

// Use your Neon pooled DATABASE_URL (already includes sslmode=require)
const pool = new Pool({
  connectionString: process.env.DATABASE_URL,
  // Neon requires TLS. If your connection string already has sslmode=require, this is safe.
  ssl: { rejectUnauthorized: false },
});

// In-memory "burn after use"
let usedOnce = false;

/**
 * Exposes GET /api/bootstrap/superadmin exactly once while BOOTSTRAP_TOKEN is set.
 * Query params:
 *   token     - must equal process.env.BOOTSTRAP_TOKEN
 *   email     - superadmin email to create/elevate
 *   password  - superadmin password to set
 *   school    - optional school name to create/use (default "System")
 *
 * On success:
 *   - Ensures a school row exists (by name)
 *   - Creates (or upgrades) the user to SUPER_ADMIN with that school_id
 *   - Logs the user into the current session (sets req.session.user)
 */
export function registerBootstrapRoute(app: Express) {
  if (!process.env.BOOTSTRAP_TOKEN) return; // only expose when explicitly enabled

  app.get("/api/bootstrap/superadmin", async (req: Request, res: Response) => {
    try {
      if (usedOnce) return res.status(410).json({ error: "bootstrap route was already used" });

      const token = String(req.query.token ?? "");
      if (!token || token !== process.env.BOOTSTRAP_TOKEN) {
        return res.status(401).json({ error: "invalid token" });
      }

      const email = String(req.query.email ?? "").toLowerCase().trim();
      const password = String(req.query.password ?? "");
      const schoolName = String(req.query.school ?? "System").trim();

      if (!email || !password) {
        return res.status(400).json({ error: "email and password are required" });
      }

      const passwordHash = await bcrypt.hash(password, 12);

      const client = await pool.connect();
      try {
        await client.query("BEGIN");

        // 1) Ensure the school exists (uses a unique name if you have a unique constraint on schools.name)
        const schoolResult = await client.query(
          `
          INSERT INTO schools (name)
          VALUES ($1)
          ON CONFLICT (name) DO UPDATE SET name = EXCLUDED.name
          RETURNING id
        `,
          [schoolName]
        );
        const schoolId = schoolResult.rows[0].id;

        // 2) Create or elevate the user to SUPER_ADMIN
        // Adjust column names if yours differ (password_hash, role, school_id are common).
        const userResult = await client.query(
          `
          INSERT INTO users (email, password_hash, role, school_id)
          VALUES ($1, $2, 'SUPER_ADMIN', $3)
          ON CONFLICT (email)
          DO UPDATE SET password_hash = EXCLUDED.password_hash,
                        role = 'SUPER_ADMIN',
                        school_id = EXCLUDED.school_id
          RETURNING id, email, role, school_id
        `,
          [email, passwordHash, schoolId]
        );
        const user = userResult.rows[0];

        await client.query("COMMIT");

        // 3) Log them in
        // If your session payload uses different keys, set them here.
        // @ts-ignore: session typing
        req.session.user = {
          id: user.id,
          email: user.email,
          role: user.role,
          schoolId: user.school_id,
        };

        usedOnce = true; // burn after success
        return res.status(201).json({ ok: true, user });
      } catch (e) {
        await client.query("ROLLBACK");
        throw e;
      } finally {
        client.release();
      }
    } catch (err: any) {
      console.error("[bootstrap] error:", err);
      return res.status(500).json({ error: "bootstrap failed", detail: err?.message ?? String(err) });
    }
  });
}
