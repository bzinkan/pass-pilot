B) Auth guard redirect loop
Cause: Client checks role, gets 401/403, immediately router.push('/login'); login page auto-redirects back → loop → blank flash.

Fix guard to show states, not instant redirects

// SuperAdminGuard.tsx
import { useEffect, useState } from 'react';
import { useRouter } from 'next/navigation';

type State = { status: 'loading'|'ok'|'forbidden'|'error'; me?: any; err?: string };

export default function SuperAdminGuard({ children }: { children: React.ReactNode }) {
  const [s, setS] = useState<State>({ status: 'loading' });
  const router = useRouter();

  useEffect(() => {
    let cancelled = false;
    (async () => {
      try {
        const res = await fetch('/api/admin/me', { credentials: 'include' });
        if (!res.ok) { setS({ status: res.status === 403 ? 'forbidden' : 'error' }); return; }
        const me = await res.json();
        if (!cancelled) setS({ status: me.role === 'SUPER_ADMIN' ? 'ok' : 'forbidden', me });
      } catch (e:any) {
        if (!cancelled) setS({ status: 'error', err: String(e) });
      }
    })();
    return () => { cancelled = true; };
  }, []);

  if (s.status === 'loading') return <div className="p-6">Checking access…</div>;
  if (s.status === 'forbidden') return <div className="p-6">No access to Super Admin.</div>;
  if (s.status === 'error') return <div className="p-6">Error loading dashboard.</div>;

  return <>{children}</>;
}
