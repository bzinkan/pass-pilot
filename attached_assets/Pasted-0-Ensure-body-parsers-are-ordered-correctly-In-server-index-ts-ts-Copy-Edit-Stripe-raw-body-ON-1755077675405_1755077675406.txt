// server/services/registration.ts
import bcrypt from "bcryptjs";
import pkg from "pg";
const { Pool } = pkg;

export const pool = new Pool({
  connectionString: process.env.DATABASE_URL,
  ssl: { rejectUnauthorized: false },
});

const FREE_SCHOOL_NAME = (process.env.FREE_SCHOOL_NAME || "").trim().toLowerCase();
const norm = (s: string) => s.trim().replace(/\s+/g, " ");

export type RegisterInput = {
  schoolName: string;
  email: string;
  password: string;
  adminName?: string;
};

export async function registerSchoolAndFirstUser(input: RegisterInput) {
  const schoolName = norm(input.schoolName);
  const email = String(input.email || "").toLowerCase().trim();
  const password = String(input.password || "");
  const adminName = String(input.adminName || "").trim();
  if (!schoolName || !email || !password) {
    const err: any = new Error("MISSING_FIELDS");
    err.http = 400;
    throw err;
  }

  const isDesignatedFree =
    FREE_SCHOOL_NAME && schoolName.toLowerCase() === FREE_SCHOOL_NAME;

  const client = await pool.connect();
  try {
    await client.query("BEGIN");

    // Block duplicates (CI)
    const dup = await client.query(
      `SELECT id, name FROM schools WHERE lower(name)=lower($1) LIMIT 1`,
      [schoolName]
    );
    if (dup.rowCount) {
      const err: any = new Error("SCHOOL_TAKEN");
      err.http = 409;
      err.extra = { school: dup.rows[0] };
      throw err;
    }

    // Insert school (rich schema if columns exist; fallback to name-only)
    let school: any;
    try {
      const s = await client.query(
        `INSERT INTO schools (name, plan, is_free, trial_end)
         VALUES ($1, $2, $3, CASE WHEN $3 THEN NULL ELSE NOW() + INTERVAL '14 days' END)
         RETURNING id, name, plan, is_free, trial_end`,
        [schoolName, isDesignatedFree ? "FREE" : "TRIAL", isDesignatedFree]
      );
      school = s.rows[0];
    } catch (e: any) {
      if (e?.code !== "42703") throw e; // undefined_column -> fallback
      const s = await client.query(
        `INSERT INTO schools (name) VALUES ($1) RETURNING id, name`,
        [schoolName]
      );
      school = s.rows[0];
    }

    // First user becomes ADMIN
    const { rows: [{ n }] } = await client.query(
      `SELECT COUNT(*)::int AS n FROM users WHERE school_id=$1`,
      [school.id]
    );
    const role = n === 0 ? "ADMIN" : "TEACHER";

    const pwHash = await bcrypt.hash(password, 12);
    const user = await client.query(
      `INSERT INTO users (email, password_hash, role, school_id, name)
       VALUES ($1, $2, $3, $4, NULLIF($5,'')) 
       RETURNING id, email, role, school_id`,
      [email, pwHash, role, school.id, adminName]
    );

    await client.query("COMMIT");
    return { school, user: user.rows[0] };
  } catch (e: any) {
    await client.query("ROLLBACK");
    if (e?.code === "23505") {
      const err: any = new Error("CONFLICT");
      err.http = 409;
      err.detail = e.detail;
      throw err;
    }
    throw e;
  } finally {
    client.release();
  }
}

// server/routes-registration-v2.ts
import type { Express, Request, Response } from "express";
import { registerSchoolAndFirstUser } from "../services/registration";

export function registerRegistrationV2(app: Express) {
  app.post("/api/registration/v2/register", async (req: Request, res: Response) => {
    try {
      const { schoolName, adminName, email, password } = req.body || {};
      const result = await registerSchoolAndFirstUser({ schoolName, adminName, email, password });
      return res.status(201).json({ ok: true, ...result });
    } catch (e: any) {
      const status = e?.http || 400;
      if (e?.message === "SCHOOL_TAKEN") return res.status(409).json({ error: "SCHOOL_TAKEN", ...(e.extra || {}) });
      if (e?.message === "MISSING_FIELDS") return res.status(400).json({ error: "MISSING_FIELDS", message: "Email, password, and school name are required" });
      return res.status(status).json({ error: e?.message || "REGISTER_FAILED", detail: e?.detail });
    }
  });
}

// server/routes-registration-compat.ts
import type { Express, Request, Response } from "express";
import { registerSchoolAndFirstUser } from "../services/registration";

/**
 * Old endpoint some UIs still call: /api/auth/register
 * Accepts many possible field names and forwards to the same service.
 */
export function registerRegistrationCompat(app: Express) {
  app.post("/api/auth/register", async (req: Request, res: Response) => {
    try {
      const b = req.body || {};
      const schoolName = b.schoolName || b.school || b.school_name || b.schoolIdName || b.school_id_name || "";
      const adminName = b.adminName || b.name || b.admin || b.administratorName || "";
      const email = (b.email || b.adminEmail || b.username || "").toString();
      const password = (b.password || b.pass || b.pw || b.secret || "").toString();

      const result = await registerSchoolAndFirstUser({ schoolName, adminName, email, password });
      return res.status(201).json({ ok: true, ...result });
    } catch (e: any) {
      const status = e?.http || 400;
      if (e?.message === "SCHOOL_TAKEN") return res.status(409).json({ error: "SCHOOL_TAKEN", ...(e.extra || {}) });
      if (e?.message === "MISSING_FIELDS") return res.status(400).json({ error: "MISSING_FIELDS", message: "Email, password, and school name are required" });
      return res.status(status).json({ error: e?.message || "REGISTER_FAILED", detail: e?.detail });
    }
  });
}

Why this fixes your “is v2 fully implemented?” issue
Even if the UI is still posting to the old route or using different field names, the compat route normalizes and calls the same core function, so no more MISSING_FIELDS.

Registration logic exists once (in services/registration.ts), so there’s no “old SQL” left to accidentally hit.

Body parsers ensure req.body is populated (the common culprit when you see “missing fields”).
